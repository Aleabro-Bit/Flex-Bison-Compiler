define hanoi(n, from_rod, to_rod, aux_rod) {
    whether [n == 1] then: {
        print(<<Move disk 1 from>>, from_rod, <<to>>, to_rod, <<#k>>);
        return;;
    }
    hanoi(n - 1, from_rod, aux_rod, to_rod);
    print(<<Move disk>>, n, <<from>>, from_rod, <<to>>, to_rod, <<#k>>);
    hanoi(n - 1, aux_rod, to_rod, from_rod);
}

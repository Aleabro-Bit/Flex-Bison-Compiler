--- set the value for the test
num n = input(<<insert a value for the test>>);

--- generates a list of numbers from 1 to n
list numbers;
from [num i = 0 to n !> 1] {
    numbers = numbers + i;
}

--- filters only prime numbers
list primes = [0];
from [num i = 0 to size(numbers)-1 !> 1] {
    num candidate = get(numbers, i);
    whether [candidate < 2] then: { continue; }

    num is_prime = 1;
    from [num j = 2 to candidate/j^2-1 !> 1] {
        num temp = candidate;
        from [num k = 0 to candidate !> 1] {
            temp = temp - j;
            whether [temp < 0] then: { break; }
            whether [temp == 0] then: { is_prime = 0; break; }
        }
        whether [is_prime == 0] then: { break; }
    }

    whether [is_prime == 1] then: {
        primes = primes + candidate;
    }
}

print(<<the prime numbers are:#k>>, primes);
